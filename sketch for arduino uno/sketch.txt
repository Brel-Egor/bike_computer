#include <LiquidCrystal_I2C.h>
#include <Wire.h>
LiquidCrystal_I2C lcd(0x27,16,2);


#define reed A0

const int buttonPin = 2;     
const int ledPin =  8;      
unsigned long currentMillis = 0;
unsigned long previousMillis = 0;
int ledState = LOW;

int buttonState = 0;          
const long interval = 350;

int reedVal;
long timer;
float mph;
float radius = 13.5;
float circumference;

int maxReedCounter = 100;
int reedCounter;


void setup(){
  lcd.init();
    // Инициализация lcd
  lcd.backlight();
  pinMode(ledPin, OUTPUT);     
  // инициализируем пин, подключенный к кнопке, как вход
  pinMode(buttonPin, INPUT);
  reedCounter = maxReedCounter;
  circumference = 2*3.14*radius;
  pinMode(reed, INPUT);
  
  
  cli();

  
  TCCR1A = 0;
  TCCR1B = 0;
  TCNT1  = 0;
 
  OCR1A = 1999;

  TCCR1B |= (1 << WGM12);
 
  TCCR1B |= (1 << CS11);   
 
  TIMSK1 |= (1 << OCIE1A);
  
  sei();
 
  
  Serial.begin(9600);
}


ISR(TIMER1_COMPA_vect) {
  reedVal = digitalRead(reed);
  if (reedVal){
    if (reedCounter == 0){
      mph = (56.8*float(circumference))/float(timer);
      timer = 0;
      reedCounter = maxReedCounter;
    }
    else{
      if (reedCounter > 0){
        reedCounter -= 1;
      }
    }
  }
  else{
    if (reedCounter > 0){
      reedCounter -= 1;
    }
  }
  if (timer > 2000){
    mph = 0;
  }
  else{
    timer += 1;
  } 
}

void displayMPH(){
 lcd.clear();
 lcd.setCursor(0, 1);
 lcd.print(mph);
}

void loop(){
  
  displayMPH();
  buttonState = digitalRead(buttonPin);
  currentMillis = millis();

  if (buttonState == HIGH) {   
     
    if (currentMillis - previousMillis >= interval) {
    
    previousMillis = currentMillis;

    
    if (ledState == LOW) {
      ledState = HIGH;
    } else {
      ledState = LOW;
    }

    digitalWrite(ledPin, ledState);
  }
  }
  else 
  {
    ledState = LOW;
    digitalWrite(ledPin, ledState);
  }
  delay(1000);
}
